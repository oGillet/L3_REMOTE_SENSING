# Load packages
library(raster)
library(rgdal)
library(ggplot2)
library(RStoolbox)

setwd(".")

# Load Landsat images
pan <- raster("01_JANVIER_2017_L8_BREST_B8_clip.tif")

lowResImg_pan

## Plot
